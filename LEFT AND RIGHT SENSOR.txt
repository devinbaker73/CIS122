//including the libraries
#include <AFMotor.h>

//defining pins and variables for the IR Sensors
#define left A0

#define right A1

//defining motors and assigning their ports in Motor Driver
AF_DCMotor motorA(A, MOTOR12_1KHZ); 

AF_DCMotor motorB(B, MOTOR12_1KHZ);

AF_DCMotor motorC(C, MOTOR34_1KHZ);

AF_DCMotor motorD(D, MOTOR34_1KHZ);

void setup() {
  //declaring pin types
  pinMode(left,INPUT);
  
  pinMode(right,INPUT);
  
  //begin serial communication
  Serial.begin(9600);
  
}
