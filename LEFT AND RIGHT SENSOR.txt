//including the libraries
#include <AFMotor.h>

//defining pins and variables for the IR Sensors
#define left A0

#define right A1

//defining motors and assigning their ports in Motor Driver
AF_DCMotor motor1(1, MOTOR12_1KHZ); 

AF_DCMotor motor2(2, MOTOR12_1KHZ);

AF_DCMotor motor3(3, MOTOR34_1KHZ);

AF_DCMotor motor4(4, MOTOR34_1KHZ);

void setup() {
  //declaring pin types
  pinMode(left,INPUT);
  
  pinMode(right,INPUT);
  
  //begin serial communication
  Serial.begin(9600);
  
}

void loop(){
  //printing values of the sensors to the serial monitor
  Serial.println(digitalRead(left));
  
  Serial.println(digitalRead(right));
//will check if the line is detected by the right sessor of the motor.
  else if(digitalRead(left)==0 && !analogRead(right)==0){
    //turn left
    motor1.run(FORWARD);
    // calls to activate motor number 1 to run backwards.
    
    motor1.setSpeed(200);
    
    motor2.run(FORWARD);
    // calls to activate motor number 2 to run backwards.
    motor2.setSpeed(200);
    // allows to set desired speed of the motor.
    
    motor3.run(BACKWARD);
    // calls to activate motor number 3 to run backwards.
    
    motor3.setSpeed(200);
    
    motor4.run(BACKWARD);
		// calls to activate motor number 4 to run backwards.
    
    motor4.setSpeed(200);

    motor1.run(FORWARD);
		// calls to activate motor number 1 to run forward.
    
    motor1.setSpeed(150);
    
    motor2.run(FORWARD);
		// calls to activate motor number 2 to run forward.
    
    motor2.setSpeed(150);
    
    motor3.run(BACKWARD);
		// calls to activate motor number 3 to run forward.
    
    motor3.setSpeed(150);
    
    motor4.run(BACKWARD);
		// calls to activate motor number 4 to run forward.
    
    motor4.setSpeed(150);
  }
  //will check if the line is detected by the left sessor of the motor.
  else if(!digitalRead(left)==0 && digitalRead(right)==0){
    //turn right
    motor1.run(BACKWARD);
    // calls to acticate back motor.
    
    motor1.setSpeed(200);
    // allows to set desired speed of the motor.
    
    motor2.run(BACKWARD);
    
    motor2.setSpeed(200);
    
    motor3.run(FORWARD);
    
    motor3.setSpeed(200);
    
    motor4.run(FORWARD);
    
    motor4.setSpeed(200);

    motor1.run(BACKWARD);
    
    motor1.setSpeed(150);
    
    motor2.run(BACKWARD);
    
    motor2.setSpeed(150);
    
    motor3.run(FORWARD);
    
    motor3.setSpeed(150);
    
    motor4.run(FORWARD);
    
    motor4.setSpeed(150);
   
  }
